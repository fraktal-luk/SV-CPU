@section prog_main

$start
	add_i r20, r0, 0x4000 ; initialize exc stack

	add_i r10, r0, 10
	lds r4, r0, 1

	add_h r2, r0, 0x10 ; db flag
	or_r r4, r4, r2


	sts r4, r0, 5
	add_i r3, r0, 0x24
	sts r3, r0, 3  ; set return address to $startsteps
	sys_reti


$startsteps
	add_i r11, r10, -5
	jz_i  r0, $L1

$L2
	add_i r12, r10, -6

$endsteps
	sys_dbcall

$L1
	jnz_i r11, $L2
